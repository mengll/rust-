### rust的泛型还是和个性的

```
struct Point<T>{
    x:T,
    y:T
}

// 在Point<T> 的解构体上实现了x的方法
impl <T> Point<T>{
    fn x(&self) ->&T{ // 返回当前类
        match &self.x {  // 数据类型的匹配
            i32 => println!("zhengshu"),
            String =>println!("自耦富川"),
            _ =>println!("未知")
        }
        &self.x
    }
}

fn main(){
    let  p =Point{x:5,y:6};
    println!("p.x = {}",p.x())
}
```
