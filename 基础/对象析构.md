### rust 中的对象，解结构赋值到，对应的参数接收位置
```
struct P(f32,f32,f32);
fn calc(arg:P)->f32 {
    let P(x,_,y) = arg;
    x*x +y*y
}

fn func(P(x,_,y):P) ->f32{
    x*x + y*y
}
// 对象的析构的方法
fn main(){
    let t = P(1.0,2.0,3.0);
   // println!("{}",calc(t));
    println!("{}",func(t)); // 一个对象，只能被一个对象，或函数使用
}

可以使用 .. 标识其它的参数  _ 占位符，无法赋值，取值
fn main(){
 let x =(12,23,34);
 let (a,..,_y) = x;
  println!("{}--{}",a,_y);
}

### match的析构的方法

fn category(x:i64){
    println!("{}",x);
    match x {
        -1=> println!("engative"),
        0  => println!("zero"),
        1 => println!("frist"),
        _ => println!("{}","This default") // match 匹配的时候default 是必须填的，不然会报错
    }
}

// 清洗当前的数据
fn main(){
    let y:i64 = 1;
    category(y);
}

// 模式匹配并满足相关的匹配的条件
fn main(){
    let x = 'a';
    match x {
        'a'...'z'  => println!("匹配到正确的值a-z"),  // 匹配一个范围 x   
        'a'...'z' if x =='b' => println!("匹配到正确的值b"),  // 匹配一个范围 x  并且满足if的条件
        _ => println!("Test")
    }
}

// 变量赋值为新额值
fn sun(x:i32,y:i32)->i32{
    x+y
}

// 新变量的声明
fn main(){
    let x = 'c';
    match x {
       e @ 'a'...'z' => println!("new value =>{}",e), // c 变量赋值为新的值
       _ => println!("default")
    }
}
```
